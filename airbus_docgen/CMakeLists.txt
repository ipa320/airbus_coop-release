project(airbus_docgen)

find_package(catkin REQUIRED COMPONENTS
  roslib
  rospy
)

catkin_python_setup()

catkin_package()

  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
)

catkin_install_python(PROGRAMS src/airbus_docgen/docgen/index.py
  DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)
